614static inline void adc_set_gain(
615 struct adc_module *const module_inst,
616 const enum adc_gain_factor gain_factor)
617{
618 /* Sanity check arguments */
619 Assert(module_inst);
620 Assert(module_inst->hw);
621
622 Adc *const adc_module = module_inst->hw;
623
624 while (adc_is_syncing(module_inst)) {
625 /* Wait for synchronization */
626 }
627
628 /* Set new gain factor */
629 adc_module->INPUTCTRL.reg =
630 (adc_module->INPUTCTRL.reg & ~ADC_INPUTCTRL_GAIN_Msk) |
631 (gain_factor);
632}

654static inline enum status_code adc_set_pin_scan_mode(
655 struct adc_module *const module_inst,
656 uint8_t inputs_to_scan,
657 const uint8_t start_offset)
658
659{
660 /* Sanity check arguments */
661 Assert(module_inst);
662 Assert(module_inst->hw);
663
664 Adc *const adc_module = module_inst->hw;
665
666 if (inputs_to_scan > 0) {
667 /*
668 * Number of input sources included is the value written to INPUTSCAN
669 * plus 1.
670 */
671 inputs_to_scan--;
672 }
673
674 if (inputs_to_scan > (ADC_INPUTCTRL_INPUTSCAN_Msk >> ADC_INPUTCTRL_INPUTSCAN_Pos) ||
675 start_offset > (ADC_INPUTCTRL_INPUTOFFSET_Msk >> ADC_INPUTCTRL_INPUTOFFSET_Pos)) {
676 /* Invalid number of input pins */
677 return STATUS_ERR_INVALID_ARG;
678 }
679
680 while (adc_is_syncing(module_inst)) {
681 /* Wait for synchronization */
682 }
683
684 /* Set pin scan mode */
685 adc_module->INPUTCTRL.reg =
686 (adc_module->INPUTCTRL.reg &
687 ~(ADC_INPUTCTRL_INPUTSCAN_Msk | ADC_INPUTCTRL_INPUTOFFSET_Msk)) |
688 (start_offset << ADC_INPUTCTRL_INPUTOFFSET_Pos) |
689 (inputs_to_scan << ADC_INPUTCTRL_INPUTSCAN_Pos);
690
691 return STATUS_OK;
692}
