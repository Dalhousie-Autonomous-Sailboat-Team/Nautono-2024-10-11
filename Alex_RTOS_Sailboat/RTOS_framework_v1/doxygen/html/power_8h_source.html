167static inline enum status_code system_set_sleepmode(
168 const enum system_sleepmode sleep_mode)
169{
170
171#if (SAMD20 || SAMD21 || SAMR21)
172
173 /* Get MCU revision */
174 uint32_t rev = DSU->DID.reg;
175
176 rev &= DSU_DID_REVISION_Msk;
177 rev = rev >> DSU_DID_REVISION_Pos;
178
179#if (SAMD20)
180 if (rev < _SYSTEM_MCU_REVISION_E) {
181 /* Errata 13140: Make sure that the Flash does not power all the way down
182 * when in sleep mode. */
183 NVMCTRL->CTRLB.bit.SLEEPPRM = NVMCTRL_CTRLB_SLEEPPRM_DISABLED_Val;
184 }
185#endif
186
187#if (SAMD21 || SAMR21)
188 if (rev < _SYSTEM_MCU_REVISION_D) {
189 /* Errata 13140: Make sure that the Flash does not power all the way down
190 * when in sleep mode. */
191 NVMCTRL->CTRLB.bit.SLEEPPRM = NVMCTRL_CTRLB_SLEEPPRM_DISABLED_Val;
192 }
193#endif
194
195#endif
196
197 switch (sleep_mode) {
198 case SYSTEM_SLEEPMODE_IDLE_0:
199 case SYSTEM_SLEEPMODE_IDLE_1:
200 case SYSTEM_SLEEPMODE_IDLE_2:
201 SCB->SCR &= ~SCB_SCR_SLEEPDEEP_Msk;
202 PM->SLEEP.reg = sleep_mode;
203 break;
204
205 case SYSTEM_SLEEPMODE_STANDBY:
206 SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
207 break;
208
209 default:
210 return STATUS_ERR_INVALID_ARG;
211 }
212
213 return STATUS_OK;
214}
