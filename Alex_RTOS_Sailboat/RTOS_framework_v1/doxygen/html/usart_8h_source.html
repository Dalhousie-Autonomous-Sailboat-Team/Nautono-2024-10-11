949static inline enum status_code usart_lock(
950 struct usart_module *const module)
951{
952 enum status_code status;
953
954 system_interrupt_enter_critical_section();
955
956 if (module->locked) {
957 status = STATUS_BUSY;
958 } else {
959 module->locked = true;

962
963 system_interrupt_leave_critical_section();
