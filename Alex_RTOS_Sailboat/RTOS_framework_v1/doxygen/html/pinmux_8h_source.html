371static inline uint8_t system_pinmux_pin_get_mux_position(
372 const uint8_t gpio_pin)
373{
374 PortGroup *const port = system_pinmux_get_group_from_gpio_pin(gpio_pin);
375 uint32_t pin_index = (gpio_pin % 32);
376
377 if (!(port->PINCFG[pin_index].reg & PORT_PINCFG_PMUXEN)) {
378 return SYSTEM_PINMUX_GPIO;
379 }
380
381 uint32_t pmux_reg = port->PMUX[pin_index / 2].reg;
382
383 if (pin_index & 1) {
384 return (pmux_reg & PORT_PMUX_PMUXO_Msk) >> PORT_PMUX_PMUXO_Pos;
385 }
386 else {
387 return (pmux_reg & PORT_PMUX_PMUXE_Msk) >> PORT_PMUX_PMUXE_Pos;
388 }
389}
