499static inline uint32_t adc_get_status(
500 struct adc_module *const module_inst)
501{
502 /* Sanity check arguments */
503 Assert(module_inst);
504 Assert(module_inst->hw);
505
506 Adc *const adc_module = module_inst->hw;
507
508 uint32_t int_flags = adc_module->INTFLAG.reg;
509
510 uint32_t status_flags = 0;
511
512 /* Check for ADC Result Ready */
513 if (int_flags & ADC_INTFLAG_RESRDY) {
514 status_flags |= ADC_STATUS_RESULT_READY;
515 }
516
517 /* Check for ADC Window Match */
518 if (int_flags & ADC_INTFLAG_WINMON) {
519 status_flags |= ADC_STATUS_WINDOW;
520 }
521
522 /* Check for ADC Overrun */
523 if (int_flags & ADC_INTFLAG_OVERRUN) {
524 status_flags |= ADC_STATUS_OVERRUN;
525 }
526
527 return status_flags;
528}

806static inline enum status_code adc_read(
807 struct adc_module *const module_inst,
808 uint16_t *result)
809{
810 Assert(module_inst);
811 Assert(module_inst->hw);
812 Assert(result);
813
814 if (!(adc_get_status(module_inst) & ADC_STATUS_RESULT_READY)) {
815 /* Result not ready */
816 return STATUS_BUSY;
817 }
818
819 Adc *const adc_module = module_inst->hw;
820
821#if (SAMD) || (SAMHA1) || (SAMHA0) || (SAMR21)
822 while (adc_is_syncing(module_inst)) {
823 /* Wait for synchronization */
824 }
825#endif
826
827 /* Get ADC result */
828 *result = adc_module->RESULT.reg;
829
830 /* Reset ready flag */
831 adc_clear_status(module_inst, ADC_STATUS_RESULT_READY);
832
833 if (adc_get_status(module_inst) & ADC_STATUS_OVERRUN) {
834 adc_clear_status(module_inst, ADC_STATUS_OVERRUN);
835 return STATUS_ERR_OVERFLOW;
836 }
837
838 return STATUS_OK;
839}
