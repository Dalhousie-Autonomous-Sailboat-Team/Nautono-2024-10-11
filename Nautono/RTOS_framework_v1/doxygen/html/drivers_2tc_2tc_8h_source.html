1136static inline void tc_enable_events(
1137 struct tc_module *const module_inst,
1138 struct tc_events *const events)
1139{
1140 /* Sanity check arguments */
1141 Assert(module_inst);
1142 Assert(module_inst->hw);
1143 Assert(events);
1144
1145 Tc *const tc_module = module_inst->hw;
1146
1147 uint32_t event_mask = 0;
1148
1149 if (events->invert_event_input == true) {
1150 event_mask |= TC_EVCTRL_TCINV;
1151 }
1152
1153 if (events->on_event_perform_action == true) {
1154 event_mask |= TC_EVCTRL_TCEI;
1155 }
1156
1157 if (events->generate_event_on_overflow == true) {
1158 event_mask |= TC_EVCTRL_OVFEO;
1159 }
1160
1161 for (uint8_t i = 0; i < NUMBER_OF_COMPARE_CAPTURE_CHANNELS; i++) {
1162 if (events->generate_event_on_compare_channel[i] == true) {
1163 event_mask |= (TC_EVCTRL_MCEO(1) << i);
1164 }
1165 }
1166
1167 tc_module->COUNT8.EVCTRL.reg |= event_mask | events->event_action;
1168}
