280static inline uint32_t rtc_tamper_get_detect_flag (struct rtc_module *const module)
281{
282 /* Sanity check arguments */
283 Assert(module);
284 Assert(module->hw);
285
286 uint32_t tamper_id = module->hw->MODE0.TAMPID.reg;
287 uint32_t detect_flags = 0;
288
289 if (tamper_id & RTC_TAMPID_TAMPID0) {
290 detect_flags |= RTC_TAMPER_DETECT_ID0;
291 }
292
293 if (tamper_id & RTC_TAMPID_TAMPID1) {
294 detect_flags |= RTC_TAMPER_DETECT_ID1;
295 }
296
297 if (tamper_id & RTC_TAMPID_TAMPID2) {
298 detect_flags |= RTC_TAMPER_DETECT_ID2;
299 }
300
301 if (tamper_id & RTC_TAMPID_TAMPID3) {
302 detect_flags |= RTC_TAMPER_DETECT_ID3;
303 }
304
305 if (tamper_id & RTC_TAMPID_TAMPID4) {
306 detect_flags |= RTC_TAMPER_DETECT_ID4;
307 }
308
309 if (tamper_id & RTC_TAMPID_TAMPEVT) {
310 detect_flags |= RTC_TAMPER_DETECT_EVT;
