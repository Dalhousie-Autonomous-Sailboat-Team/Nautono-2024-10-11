313static inline enum status_code i2c_master_lock(
314 struct i2c_master_module *const module)
315{
316 enum status_code status;
317
318 system_interrupt_enter_critical_section();
319
320 if (module->locked) {
321 status = STATUS_BUSY;
322 } else {
323 module->locked = true;
324 status = STATUS_OK;
325 }
326
327 system_interrupt_leave_critical_section();
328
329 return status;
330}

366static inline bool i2c_master_is_syncing (
367 const struct i2c_master_module *const module)
368{
369 /* Sanity check */
370 Assert(module);
371 Assert(module->hw);
372
373 SercomI2cm *const i2c_hw = &(module->hw->I2CM);
374
375#if defined(FEATURE_SERCOM_SYNCBUSY_SCHEME_VERSION_1)
376 return (i2c_hw->STATUS.reg & SERCOM_I2CM_STATUS_SYNCBUSY);
377#elif defined(FEATURE_SERCOM_SYNCBUSY_SCHEME_VERSION_2)
378 return (i2c_hw->SYNCBUSY.reg & SERCOM_I2CM_SYNCBUSY_MASK);
379#else
380# error Unknown SERCOM SYNCBUSY scheme!
381#endif
382}

471static inline void i2c_master_enable(
472 const struct i2c_master_module *const module)
473{
474 /* Sanity check of arguments */
475 Assert(module);
476 Assert(module->hw);
477
478 SercomI2cm *const i2c_module = &(module->hw->I2CM);
479
480 /* Timeout counter used to force bus state */
481 uint32_t timeout_counter = 0;
482
483 /* Wait for module to sync */
484 _i2c_master_wait_for_sync(module);
485
486 /* Enable module */
487 i2c_module->CTRLA.reg |= SERCOM_I2CM_CTRLA_ENABLE;
488
489#if I2C_MASTER_CALLBACK_MODE == true
490 /* Enable module interrupts */
491 system_interrupt_enable(_sercom_get_interrupt_vector(module->hw));
492#endif
493 /* Start timeout if bus state is unknown */
494 while (!(i2c_module->STATUS.reg & SERCOM_I2CM_STATUS_BUSSTATE(1))) {
495 timeout_counter++;
496 if(timeout_counter >= (module->unknown_bus_state_timeout)) {
497 /* Timeout, force bus state to idle */
498 i2c_module->STATUS.reg = SERCOM_I2CM_STATUS_BUSSTATE(1);
499 /* Workaround #1 */
500 return;
501 }
502 }
503}

512static inline void i2c_master_disable(
513 const struct i2c_master_module *const module)
514{
515 /* Sanity check of arguments */
516 Assert(module);
517 Assert(module->hw);
518
519 SercomI2cm *const i2c_module = &(module->hw->I2CM);
520
521#if I2C_MASTER_CALLBACK_MODE == true
522 /* Disable module interrupts */
523 system_interrupt_disable(_sercom_get_interrupt_vector(module->hw));
524#endif
525
526 /* Wait for module to sync */
527 _i2c_master_wait_for_sync(module);
528
529 /* Disbale interrupt */
530 i2c_module->INTENCLR.reg = SERCOM_I2CM_INTENCLR_MASK;
531 /* Clear interrupt flag */
532 i2c_module->INTFLAG.reg = SERCOM_I2CM_INTFLAG_MASK;
533
534 /* Disable module */
535 i2c_module->CTRLA.reg &= ~SERCOM_I2CM_CTRLA_ENABLE;
536
537}
