637static inline enum status_code port_input_event_set_config(
638 const enum port_input_event n,
639 struct port_input_event_config *const config)
640{
641 Assert(config);
642 PortGroup *const port_base = port_get_group_from_gpio_pin(config->gpio_pin);
643 uint8_t pin_index = config->gpio_pin % 32;
644 struct port_config pin_conf;
645
646 port_get_config_defaults(&pin_conf);
647 /* Configure the GPIO pin as outputs*/
648 pin_conf.direction = PORT_PIN_DIR_OUTPUT;
649 port_pin_set_config(config->gpio_pin, &pin_conf);
650
651 switch (n) {
652 case PORT_INPUT_EVENT_0:
653 port_base->EVCTRL.reg |= PORT_EVCTRL_EVACT0(config->action)
654 | PORT_EVCTRL_PID0(pin_index);
655 break;
656 case PORT_INPUT_EVENT_1:
657 port_base->EVCTRL.reg |= PORT_EVCTRL_EVACT1(config->action)
658 | PORT_EVCTRL_PID1(pin_index);
659 break;
660 case PORT_INPUT_EVENT_2:
661 port_base->EVCTRL.reg |= PORT_EVCTRL_EVACT2(config->action)
662 | PORT_EVCTRL_PID2(pin_index);
663 break;
664 case PORT_INPUT_EVENT_3:
665 port_base->EVCTRL.reg |= PORT_EVCTRL_EVACT3(config->action)
666 | PORT_EVCTRL_PID3(pin_index);
667 break;
668 default:
669 Assert(false);
670 return STATUS_ERR_INVALID_ARG;
671 }
672 return STATUS_OK;
