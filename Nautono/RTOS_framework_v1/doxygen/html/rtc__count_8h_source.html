984static inline void rtc_count_enable_events(
985 struct rtc_module *const module,
986 struct rtc_count_events *const events)
987{
988 /* Sanity check arguments */
989 Assert(module);
990 Assert(module->hw);
991
992 Rtc *const rtc_module = module->hw;
993
994 uint32_t event_mask = 0;
995
996 /* Check if the user has requested an overflow event */
997 if (events->generate_event_on_overflow) {
998 event_mask |= RTC_MODE0_EVCTRL_OVFEO;
999 }
1000
1001 /* Check if the user has requested any compare events */
1002 for (uint8_t i = 0; i < RTC_NUM_OF_COMP16; i++) {
1003 if (events->generate_event_on_compare[i]) {
1004 event_mask |= RTC_MODE0_EVCTRL_CMPEO(1 << i);
1005 }
1006 }
1007
1008 /* Check if the user has requested any periodic events */
1009 for (uint8_t i = 0; i < 8; i++) {
1010 if (events->generate_event_on_periodic[i]) {
1011 event_mask |= RTC_MODE0_EVCTRL_PEREO(1 << i);
1012 }
1013 }
1014
1015#ifdef FEATURE_RTC_TAMPER_DETECTION
1016 /* Check if the user has requested a tamper event output. */
1017 if (events->generate_event_on_tamper) {
1018 event_mask |= RTC_MODE0_EVCTRL_TAMPEREO;
1019 }
1020
1021 /* Check if the user has requested a tamper event input. */
1022 if (events->on_event_to_tamper) {
1023 event_mask |= RTC_MODE0_EVCTRL_TAMPEVEI;
1024 }
1025#endif
1026
1027 /* Enable given event(s). */
1028 rtc_module->MODE0.EVCTRL.reg |= event_mask;
1029}
